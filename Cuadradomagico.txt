#include <iostream>
#include <vector>

using namespace std;

void generarCuadradoMagico(vector<vector<int>>& matriz, int n) {
    int fila = n - 1;
    int columna = n / 2;

    for (int num = 1; num <= n * n; num++) {
        matriz[fila][columna] = num;
        fila++;
        columna++;

        if (num % n == 0) {
            fila -= 2;
            columna--;
        }
        else {
            if (fila == n)
                fila = 0;
            if (columna == n)
                columna = 0;
        }
    }
}

int main() {
    int n;
    cout << "Ingrese el valor de n (debe ser impar): ";
    cin >> n;

    if (n % 2 == 0) {
        cout << "El valor de n debe ser impar." << endl;
        return 1;
    }

    vector<vector<int>> cuadrado_magico(n, vector<int>(n, 0));

    generarCuadradoMagico(cuadrado_magico, n);

    cout << "Cuadrado MÃ¡gico " << n << "x" << n << ":" << endl;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            cout << cuadrado_magico[i][j] << "\t";
        }
        cout << endl;
    }

    return 0;
}
